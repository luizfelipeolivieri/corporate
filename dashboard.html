import { app } from "../firebase/config.js"; import { getAuth, signInWithEmailAndPassword } from "https://www.gstatic.com/firebasejs/10.12.2/firebase-auth.js"; const auth = getAuth(app); document.getElementById("loginBtn").addEventListener("click", ()=>{ const email = document.getElementById("email").value; const password = document.getElementById("password").value; signInWithEmailAndPassword(auth,email,password) .then(()=>{ window.location.href="dashboard.html"; }) .catch(e=>alert(e.message)); });
